def get_latest_log_file():
    """РќР°С…РѕРґРёС‚ СЃР°РјС‹Р№ РЅРѕРІС‹Р№ Р»РѕРі-С„Р°Р№Р» РїРѕ РїР°С‚С‚РµСЂРЅСѓ"""

    # РЎРѕСЂС‚РёСЂСѓРµРј РїРѕ РІСЂРµРјРµРЅРё РјРѕРґРёС„РёРєР°С†РёРё

            data = f.read().strip().split('\n')
            if len(data) >= 2:
                return data[0], int(data[1])
            elif len(data) == 1:
                return data[0], 0
    except:
        pass
    
    return None, 0

def save_state(filepath, position):
    """РЎРѕС…СЂР°РЅСЏРµС‚ СЃРѕСЃС‚РѕСЏРЅРёРµ"""

    """РћС‚РїСЂР°РІР»СЏРµС‚ СЃС‚СЂРѕРєРё РЅР° webhook"""

            'count': len(lines),
            'logs': lines
        }

            WEBHOOK_URL,
            json=payload,
            headers={'Content-Type': 'application/json'},
            timeout=10

def monitor_logs():

    print(f"Р”РёСЂРµРєС‚РѕСЂРёСЏ Р»РѕРіРѕРІ: {LOGS_DIR}")
    print(f"Webhook: {WEBHOOK_URL}")

            current_file = get_latest_log_file()
            
            if not current_file:
                print("вљ  Р›РѕРі-С„Р°Р№Р»С‹ РЅРµ РЅР°Р№РґРµРЅС‹, РѕР¶РёРґР°РЅРёРµ...")

            file_size = os.path.getsize(current_file)

            if file_size < last_position:

            # Р•СЃР»Рё РµСЃС‚СЊ РЅРѕРІС‹Рµ РґР°РЅРЅС‹Рµ
            if file_size > last_position:
                with open(current_file, 'r', encoding='utf-8', errors='ignore') as f:
                    f.seek(last_position)
                    new_lines = []
                    
                    for line in f:
                        line = line.rstrip('\n\r')
                        if line:  # РџСЂРѕРїСѓСЃРєР°РµРј РїСѓСЃС‚С‹Рµ СЃС‚СЂРѕРєРё

        except Exception as e:
            print(f"вњ— РћС€РёР±РєР°: {e}")

if __name__ == '__main__':
